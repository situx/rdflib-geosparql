466def azimuth(a: Literal) -> Literal:
467 thegeom, thegeomsrs = LiteralUtils.processLiteralTypeToGeom(a)
468 minrect = thegeom.minimum_rotated_rectangle
469 minrectb = minrect.boundary
470 coords = [c for c in minrectb.coords] # List the line coordinates
471 segments = [shapely.geometry.LineString([a, b]) for a, b in zip(coords, coords[1:])] # Create the four side lines.
472 longest_segment = max(segments, key=lambda x: x.length)
473 p1, p2 = [c for c in longest_segment.coords] # List the start and end coordinates of it
